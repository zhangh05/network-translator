!
! Cisco ASA — NAT server/port forwarding
! Source: public_doc
! Domain: firewall, Features: nat, interface, static_route
!
interface GigabitEthernet0/0
 nameif inside
 security-level 100
 ip address 10.0.0.1 255.255.255.0
 no shutdown
!
interface GigabitEthernet0/1
 nameif outside
 security-level 0
 ip address 198.18.1.1 255.255.255.0
 no shutdown
!
object network WEB_SERVER
 host 10.0.0.10
!
object network MAIL_SERVER
 host 10.0.0.11
!
object network OUTSIDE_IPS
 range 198.18.10.100 198.18.10.110
!
nat (inside,outside) source static WEB_SERVER interface service tcp www www
nat (inside,outside) source static MAIL_SERVER interface service tcp smtp smtp
nat (inside,outside) source dynamic INSIDE_NET OUTSIDE_IPS
!
access-list OUTSIDE_IN extended permit tcp any interface outside eq www
access-list OUTSIDE_IN extended permit tcp any interface outside eq smtp
access-group OUTSIDE_IN in interface outside
!
route outside 0.0.0.0 0.0.0.0 198.18.1.254 1
!
